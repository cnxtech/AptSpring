60    public void inspectDefinitionGraph(Set<DefinitionModel> definitionGraphHeads,
61        Consumer<ErrorModel> errorListener, AssignabilityUtils assignabilityUtils, DefinitionModelStore store) {
62      for (DefinitionModel definition : definitionGraphHeads) {
63        depthFirstExpectedsInspector(definition, errorListener, assignabilityUtils, store);
64      }
65    }

67    public InstanceModel getOneWithSourceElementElseAny(final Collection<InstanceModel> possibilities) {
68      return possibilities.stream()
69           .filter(im -> im.getSourceElement().isPresent())
70           .findAny().orElseGet(() -> possibilities.iterator().next());
71    }

73    public boolean verifiedShas(DefinitionModel model, DefinitionModelStore store) {
74      boolean verified = true;
75      for (Entry<String, String> depNameToSha256 : model.getDependencyNameToSha256().entrySet()) {
76        DefinitionModel dep = store.lookup(depNameToSha256.getKey()).get(0);
77        if (model.getDependencyNameToSha256().get(dep.getIdentity()) != null) {
78          verified = verified && verifiedShas(dep, store); 
79          verified = verified && model.getDependencyNameToSha256().get(dep.getIdentity()).equals(dep.getSha256());
80        }
81      }
82      return verified;
83    }

86    /**
87     * Verify that all expected entities are marked expected.
88     * Verify that the types of provided entities satisfy all expected types....
89     * <p/>
90     * if a child record was freshly process, so too must all parent ( if the md5 has changed ).
91     *  
92     * @param definition the head of current tree of definitions linked by imports
93     * @param errorListener registers all errors found 
94     */
95    private boolean depthFirstExpectedsInspector(DefinitionModel definition, Consumer<ErrorModel> errorListener,
96        AssignabilityUtils assignabilityUtils, DefinitionModelStore store) {
97      boolean errored = false;
98      for (DefinitionModel dependency : definition.getDependencies()) {
99        if (!dependency.isLockedAnalyzed()) {
100         errored = depthFirstExpectedsInspector(dependency, errorListener, assignabilityUtils, store) || errored;
101         if (errored) {
102           return true; // no need to continue.
103         }
104       } else {
105         if (!verifiedShas(dependency, store)) {
106           errored = true;
107           errorListener.accept(
108               new ErrorModel(ErrorType.DEPENDENCY_SHA_MISMATCH, Arrays.asList(dependency), Arrays.asList(definition)));
109         }
110       }
111     }
112     
113     if (errored) {
114       return errored;
115     }
116     
117     //check that each object has only one source ( could be the imported from a dependency, in a diamond pattern )
118     //validate the expected beans are correct.
119     Map<String, InstanceModel> resolvedInstances = ensureSingleInstanceOfEachName(definition, errorListener);
120     
121     //short circuit if dependencies couldn't be resolved.
122     if (resolvedInstances == null) {
123       return false;
124     }
125     
126     //looks for cycles and unexpected missing entities.
127     errored = detectCyclesInEntityGraph(definition, resolvedInstances, errorListener);
128 
129     //check types of non-expected dependencies
130     errored = checkInstancesTypesInDefinition(definition, resolvedInstances, errorListener, assignabilityUtils) || errored;
131 
132     //check all definitions with expected, that each instance expecting a definition can use the supplied.
133     errored = checkProvidedSupplyCorrectTypes(definition, resolvedInstances, errorListener, assignabilityUtils) || errored;
134 
135     //prune all edged from graph that don't end in an expected
136 
137     //store computed expects with all types that must satisfied.
138     
139     
140     
141     if (!errored) {
142       //store as provided dependencies
143       definition.addAllProvidedInstances(resolvedInstances.values());
144       for (DefinitionModel dep : definition.getDependencies()) {
145         definition.addDependencyNameToSha256(dep.getIdentity(), dep.getSha256());
146       }
147       //storing will lock a the definition, as will reading.
148       if (!store.store(definition)) {
149         errorListener.accept(
150             new ErrorModel(ErrorType.COULD_NOT_STORE, Arrays.asList(definition), Arrays.asList(definition)));
151       }
152     }
153     
154     return errored;
155   }

157   private ErrorModel errorForMismatchedExpected(final DefinitionModel definition, ExpectedModel computedExpected,
158       final Map<String, InstanceModel> nameToEntity, AssignabilityUtils assignabilityUtils) {
159     //that which is to fill all the expectedInstance references.
160     InstanceModel providedInstance = nameToEntity.get(computedExpected.getIdentity());
161     
162     //the top level definition model should have declared this an expectedBean - maybe move that error here?
163     if (providedInstance == null) {
164       return null;
165     }
166     
167     //Instances which expect to use the provided instance
168     List<InstanceModel> mistmatchedExpectantEntities = computedExpected.getDefinitionReferenceToType().keySet().stream()
169         .map(name -> nameToEntity.get(name))
170         .filter(expectantInstance -> !assignabilityUtils.isAssignableFrom(providedInstance, expectantInstance))
171         .collect(Collectors.toList());
172     
173     if (mistmatchedExpectantEntities.size() == 0) {
174       return null;
175     } else {
176       List<InstanceModel> involved = new ArrayList<>();
177       involved.add(providedInstance);
178       involved.addAll(mistmatchedExpectantEntities);
179       return new ErrorModel(ErrorType.UNMATCHED_TYPES, involved, Arrays.asList(definition, providedInstance));
180     }
181   }

183   private boolean checkProvidedSupplyCorrectTypes(final DefinitionModel definition, final Map<String, InstanceModel> nameToEntity,
184       final Consumer<ErrorModel> errorListner, AssignabilityUtils assignabilityUtils) {
185     List<ErrorModel> errors = definition.getDependencies().stream()
186         .flatMap(dependency -> dependency.getComputedExpected().stream())
187         .filter(expectedModel -> !definition.getExpectedDefinitions().stream()
188             .anyMatch(ed -> ed.getIdentity().equals(expectedModel.getIdentity())))
189         .map(em -> errorForMismatchedExpected(definition, em, nameToEntity, assignabilityUtils))
190         .filter(errorModel -> errorModel != null)
191         .collect(Collectors.toList());
192     errors.stream().forEach(errorListner);
193     return errors.size() > 0;
194   }

197   private List<Entry<String, InstanceModel>> getEntryListForNameAndAlias(InstanceModel instanceModel) {
198     List<Entry<String, InstanceModel>> output = new ArrayList<>();
199     output.add(new SimpleEntry<>(instanceModel.getIdentity(), instanceModel));
200     for (String alias : instanceModel.getAliases()) {
201       if (!alias.equals(instanceModel.getIdentity())) {
202         output.add(new SimpleEntry<>(alias, instanceModel));
203       }
204     }
205     return output;
206   }

208   /**
209    * Verify that a single named instance model is correctly identifiable composed of owningDefinition/ElementLocation/Identity.
210    * 
211    * 
212    * @param definition the definition of 
213    * @param errorListener accepts and displays all errors produced by analyzing the models.
214    * @return returns a map of each instance name to the single model that the has been resolved to. 
215    */
216   private Map<String, InstanceModel> ensureSingleInstanceOfEachName(DefinitionModel definition,
217       Consumer<ErrorModel> errorListener) {
218     Stream<InstanceModel> imported = definition.getDependencies().stream()
219         .map(d -> d.getProvidedInstances()).flatMap(x -> x.stream());
220     Stream<InstanceModel> instanceModelStream = Stream.concat(definition.getObjectDefinitions().stream(), imported);
221 
222     Map<String, List<InstanceModel>> instancesByName = instanceModelStream
223         .flatMap(instance -> getEntryListForNameAndAlias(instance).stream())
224         .collect(Collectors.groupingBy(entry -> entry.getKey(), 
225             Collectors.mapping(entry -> entry.getValue(), Collectors.toList())));
226     
227     final Map<String, InstanceModel> resolvedDependencies = new HashMap<>();
228     boolean errored = false;
229     for (Entry<String, List<InstanceModel>> entry : instancesByName.entrySet()) {
230       Map<String, List<InstanceModel>> collidingInstances = entry.getValue().stream()
231           .collect(Collectors.groupingBy(
232               im -> im.getOwningDefinition() 
233                   + "::" 
234                   + im.getElementLocation()
235                   + "::"
236                   + im.getIdentity()
237              ));
238       if (collidingInstances.size() == 1) {
239         //there is only one entity and it is now resolved
240         InstanceModel instance = getOneWithSourceElementElseAny(collidingInstances.values().iterator().next());
241         resolvedDependencies.put(entry.getKey(), instance);
242       }
243       if (collidingInstances.size() > 1) {
244         errored = true;
245         List<AbstractModel> causes = new ArrayList<>();
246         List<AbstractModel> involved = new ArrayList<>();
247         involved.add(definition);
248         collidingInstances.values().stream().forEach(sameIdentities ->  {
249           InstanceModel instance = getOneWithSourceElementElseAny(sameIdentities);
250           causes.add(instance);
251           if (instance.getSourceElement().isPresent()) {
252             involved.add(instance);
253           }
254         });
255         errored = true;
256         errorListener.accept(new ErrorModel(ErrorType.DUPLICATE_OBJECT_DEFINITIONS, causes, involved));
257       }
258     }
259     return errored ? null : resolvedDependencies;
260   }

262   /**
263    * Inspects the instance graph for cycles, any cycle is printed as an error.   The nameToEntity parameter doesn't list expected
264    * instances, any instances that are not found in the nameToInstances map (they are looked for because they are referenced as a
265    * dependency by an instance in the map) and are not found by name in the definition's expectedInstances are treated as errors
266    * as well.
267    * 
268    * @param definition definition being processed.  Will uses it's expected list, any instances references as dependencies but
269    *     not found, not listed as expected in this DefinitionModel, will be treated as errors.
270    * @param nameToEntity name to unique instanceModels, verified before call.
271    * @param errorListner accepts and displays all errors produced by analyzing the models
272    * @return true if an error occurred, false otherwise
273    */
274   private boolean detectCyclesInEntityGraph(final DefinitionModel definition, final Map<String, InstanceModel> nameToEntity,
275       final Consumer<ErrorModel> errorListener) {
276     final Map<String, ExpectedModel> missing = new HashMap<>();
277     final DirectedGraph<BaseInstanceModel, DefaultEdge> entityGraph = new DefaultDirectedGraph<>(DefaultEdge.class);
278     for (BaseInstanceModel entity : nameToEntity.values()) {
279       if (!entityGraph.containsVertex(entity)) {
280         entityGraph.addVertex(entity);
281       }
282       if (InstanceModel.class.isAssignableFrom(entity.getClass())) {
283         InstanceModel instanceModel = (InstanceModel) entity;
284         for (InstanceDependencyModel instanceDependency : instanceModel.getDependencies()) {
285           BaseInstanceModel dependency = nameToEntity.get(instanceDependency.getIdentity());
286           if (dependency == null) {
287             dependency = missing.get(instanceDependency.getIdentity());
288             if (dependency == null) {
289               ExpectedModel expected = new ExpectedModel(instanceDependency.getIdentity());
290               dependency = expected;
291               missing.put(instanceDependency.getIdentity(), expected);
292             }
293             missing.get(instanceDependency.getIdentity())
294               .addDefinitionReferenceToType(instanceModel.getIdentity(), instanceDependency.getType());
295           }
296           if (!entityGraph.containsVertex(dependency)) {
297             entityGraph.addVertex(dependency);
298           }
299           entityGraph.addEdge(entity, dependency);
300         }
301       }
302     }
303     
304     boolean errored = errorsForCycles(errorListener, entityGraph);
305     errored = testAllMissingEntitiesAreExpected(definition, errorListener, missing, entityGraph) || errored;
306     errored = errorUnusedExpectedsOnDefinition(definition, errorListener, missing) || errored;
307     return errored;
308   }

310   private boolean checkInstancesTypesInDefinition(final DefinitionModel definition, final Map<String, InstanceModel> nameToEntity,
311       final Consumer<ErrorModel> errorListner, AssignabilityUtils assignabilityUtils) {
312     boolean errored = false;
313     List<InstanceModel> instances = definition.getObjectDefinitions();
314     for (InstanceModel instance : instances) {
315       for (InstanceDependencyModel instanceDependency : instance.getDependencies()) {
316         InstanceModel dependency = nameToEntity.get(instanceDependency.getIdentity());
317         if (dependency != null && !assignabilityUtils.isAssignableFrom(dependency, instance)) {
318           errored = true;
319           errorListner.accept(new ErrorModel(ErrorType.UNMATCHED_TYPES,
320               Arrays.asList(instance, dependency), Arrays.asList(instance, dependency)));
321         }
322       }
323     }
324     return errored;
325   }

327   private boolean errorUnusedExpectedsOnDefinition(final DefinitionModel definition, final Consumer<ErrorModel> errorListner,
328       final Map<String, ExpectedModel> missing) {
329     boolean errored = false;
330     for (ExpectedModel expected : definition.getExpectedDefinitions()) {
331       if (!missing.keySet().contains(expected.getIdentity())) {
332         errorListner.accept(new ErrorModel(ErrorType.UNUSED_EXPECTED, Arrays.asList(expected), Arrays.asList(definition)));
333         errored = true;
334       }
335     }
336     return errored;
337   }

339   private boolean errorsForCycles(final Consumer<ErrorModel> errorListner,
340       final DirectedGraph<BaseInstanceModel, DefaultEdge> entityGraph) {
341     SzwarcfiterLauerSimpleCycles<BaseInstanceModel, DefaultEdge> cycleFind = new SzwarcfiterLauerSimpleCycles<>();
342     boolean errored = false;
343     cycleFind.setGraph(entityGraph);
344     List<List<BaseInstanceModel>> cycles = cycleFind.findSimpleCycles();
345     for (List<BaseInstanceModel> cycle : cycles) {
346       errored = true;
347       errorListner.accept(new ErrorModel(ErrorType.CYCLE_IN_DEFINITION_SOURCES, cycle, cycle));
348     }
349     return errored;
350   }

352   private boolean testAllMissingEntitiesAreExpected(final DefinitionModel definition, final Consumer<ErrorModel> errorListner,
353       final Map<String, ExpectedModel> missing, final DirectedGraph<BaseInstanceModel, DefaultEdge> entityGraph) {
354     //check computed expected are actually expected
355     boolean errored = false;
356     List<String> expectedMissing = definition.getExpectedDefinitions().stream().map(em -> em.getIdentity())
357         .collect(Collectors.toList());
358     for (ExpectedModel expected : missing.values()) {
359       if (!expectedMissing.contains(expected.getIdentity())) {
360         List<AbstractModel> dependsOnMissing = Stream.concat(
361             Stream.of(definition), 
362             entityGraph.incomingEdgesOf(expected).stream().map(edge -> entityGraph.getEdgeSource(edge))
363               .filter(m -> m.getSourceElement().isPresent()))
364             .collect(Collectors.toList());
365         errored = true;
366         errorListner.accept(new ErrorModel(ErrorType.MISSING_BEAN_DEFINITIONS, Arrays.asList(expected), dependsOnMissing));
367       } else {
368         definition.addComputedExpected(expected);
369       }
370     }
371     return errored;
372   }
