31  import java.io.File;
32  import java.util.Arrays;
33  
34  import org.apache.maven.execution.MavenSession;
35  import org.apache.maven.project.MavenProject;
36  import org.codehaus.plexus.util.xml.Xpp3Dom;
37  import org.junit.Rule;
38  import org.junit.Test;
39  import org.junit.runner.RunWith;
40  import org.junit.runners.Parameterized;
41  import org.junit.runners.Parameterized.Parameters;
42  import org.springframework.beans.factory.config.BeanDefinition;
43  import org.springframework.context.annotation.Bean;
44  import org.springframework.core.SpringVersion;
45  
46  import com.salesforce.aptspring.Verified;
47  import com.salesforce.aptspring.processor.SpringAnnotationParser;
48  
49  import io.takari.maven.testing.TestMavenRuntime;
50  import io.takari.maven.testing.TestResources;
51  
52  @RunWith(Parameterized.class)
53  public class TakariIncrementalCompileTests {
54  
55    private static final String PROC = "proc";
56  
57    @Rule
58    public final TestResources resources = new TestResources();
59  
60    @Rule
61    public final TestMavenRuntime maven = new TestMavenRuntime();
62  
63    protected final String compilerId;
64  
65    public TakariIncrementalCompileTests(String compilerId) {
66      this.compilerId = compilerId;
67    }
68  
69    @Parameters(name = "{0}")
70    public static Iterable<Object[]> compilers() {
71      return Arrays.<Object[]>asList(new Object[] { "javac" }, new Object[] { "forked-javac" }, new Object[] { "jdt" });
72    }
73  
74    private Xpp3Dom newParameter(String name, String value) {
75      Xpp3Dom child = new Xpp3Dom(name);
76      child.setValue(value);
77      return child;
78    }
79  
80    @Test
81    public void testBasic() throws Exception {
82  
83      // this creates a temporary copy of src/test/projects/basic test project
84      File basedir = resources.getBasedir("basic");
85  
86      // create MavenProject model for the test project
87      MavenProject project = maven.readMavenProject(basedir);
88  
89      // add annotation processor to the test project dependencies (i.e.
90      // classpath)
91      File springContext = new File(Bean.class.getProtectionDomain().getCodeSource().getLocation().toURI());
92      maven.newDependency(springContext.getCanonicalFile()).setArtifactId("spring-context").addTo(project);
93  
94      File springCore = new File(SpringVersion.class.getProtectionDomain().getCodeSource().getLocation().toURI());
95      maven.newDependency(springCore.getCanonicalFile()).setArtifactId("spring-core").addTo(project);
96  
97      File springBean = new File(BeanDefinition.class.getProtectionDomain().getCodeSource().getLocation().toURI());
98      maven.newDependency(springBean.getCanonicalFile()).setArtifactId("spring-bean").addTo(project).addTo(project);
99  
100     File verifiedApi = new File(Verified.class.getProtectionDomain().getCodeSource().getLocation().toURI());
101     maven.newDependency(verifiedApi.getCanonicalFile()).setArtifactId("verifiedApi").addTo(project);
102 
103     File verifiedImpl = new File(
104         SpringAnnotationParser.class.getProtectionDomain().getCodeSource().getLocation().toURI());
105     maven.newDependency(verifiedImpl.getCanonicalFile()).setArtifactId("verifiedProcessor").addTo(project);
106 
107     MavenSession session = maven.newMavenSession(project);
108 
109     // run java compiler with annotation processing enabled
110     maven.executeMojo(session, project, "compile", newParameter("compilerId", compilerId), newParameter(PROC, PROC));
111     File compiledClasses = new File(basedir, "target/classes");
112     File generatedSources = new File(basedir, "target/generated-sources/annotations");
113 
114     File configurationSourceFile = (new File(generatedSources,
115         "com/salesforce/aptspring/ComputerHardwareConfiguration_forceInjectData.java"));
116     assertThat(configurationSourceFile).exists().canRead();
117 
118     File configurationClass = (new File(compiledClasses,
119         "com/salesforce/aptspring/ComputerHardwareConfiguration_forceInjectData.class"));
120     assertThat(configurationClass).exists().canRead();
121 
122     maven.executeMojo(session, project, "testCompile", newParameter("compilerId", compilerId),
123         newParameter(PROC, PROC));
124     File compiledTestClasses = new File(basedir, "target/test-classes");
125     File configurationTestClass = (new File(compiledTestClasses,
126         "com/salesforce/aptspring/RootApplicationConfiguration_forceInjectData.class"));
127     assertThat(configurationTestClass).exists().canRead();
128   }
129 }
